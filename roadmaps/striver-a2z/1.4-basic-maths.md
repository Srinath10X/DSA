## Count Digits
```cpp
#include <iostream>

int main() {
    int n = 1234;
    int count = 0;

    while (n != 0) {
        n /= 10;
        count++;
    }

    std::cout << "Number of digits: " << count << std::endl;
}
```

## Reverse a number
```cpp
#include <iostream>

int main() {
    int n = 1234;
    int original = n, reverse = 0;

    while (n != 0) {
        int last_digit = n % 10;
        reverse = reverse * 10 + last_digit;
        n /= 10;
    }
    return 0;
}
```

## Check Palindrome
```cpp
#include <iostream>

int main() {
    int n = 1234;
    int original = n, reverse = 0;

    while (n != 0) {
        int last_digit = n % 10;
        reverse = reverse * 10 + last_digit;
        n /= 10;
    }

    if (original == reverse) {
        std::cout << "Palindrome" << std::endl;
    } else {
        std::cout << "Not Palindrome" << std::endl;
    }
    return 0;
}
```
