## Count Digits
```cpp
#include <iostream>

int main() {
    int n = 1234;
    int count = 0;

    while (n != 0) {
        n /= 10;
        count++;
    }

    std::cout << "Number of digits: " << count << std::endl;
}
```

## Reverse a number
```cpp
#include <iostream>

int main() {
    int n = 1234;
    int original = n, reverse = 0;

    while (n != 0) {
        int last_digit = n % 10;
        reverse = reverse * 10 + last_digit;
        n /= 10;
    }
    return 0;
}
```

## Check Palindrome
```cpp
#include <iostream>

int main() {
    int n = 1234;
    int original = n, reverse = 0;

    while (n != 0) {
        int last_digit = n % 10;
        reverse = reverse * 10 + last_digit;
        n /= 10;
    }

    if (original == reverse) {
        std::cout << "Palindrome" << std::endl;
    } else {
        std::cout << "Not Palindrome" << std::endl;
    }
    return 0;
}
```
## GCD
```cpp
#include <iostream>

/// Bruteforce 
int min(int a, int b) {
    return a < b ? a : b;
}

int gcdBrute(int a, int b) {
    int gcd = 1;

    for (int i = 1; i < min(a, b); ++i) {
        if (a % i == 0 && b % i == 0) {
            gcd = i;
        }
    }

    return gcd;
}

/// Optimal
int gcdOptimal(int a, int b) {
    while (a > 0 && b > 0) {
        if (a > b) {
            a = a % b;
        } else {
            b = b % a;
        }
    }

    return a == 0 ? b : a;
}

/// Recursive
int gcdRecursive(int a, int b) {
    return a == 0 ? b : gcd(b % a, a);
}
```

## Armstrong Number
```cpp
#include <iostream>

int main() {
    int num = 153;
    int original = num;
    int sum = 0;

    while(num != 0) {
        int last_digit = num % 10;
        num /= 10;
        sum += last_digit * last_digit * last_digit;
    }

    if (sum == original) {
        std::cout << "Armstrong" << std::endl;
    } else {
        std::cout << "Not Armstrong" << std::endl;
    }

    return 0;
}
```
